let todos = [];
let running = true;

alert("Welcome to Todo App!\nCommands: add, list, delete, quit");

while (running) {
  let command = prompt("Enter command (add / list / delete / quit):");

  if (command === null) {
    alert("Goodbye!");
    break;
  }

  command = command.toLowerCase().trim();

  if (command === "add") {
    let task = prompt("Enter task to add:");
    if (task && task.trim() !== "") {
      todos.push(task.trim());
      alert("Task added!");
      console.log("Added:", task);
    } else {
      alert("Task cannot be empty.");
    }

  } else if (command === "list") {
    if (todos.length === 0) {
      alert("No tasks yet.");
      console.log("Todo list is empty.");
    } else {
      let list = "Your Tasks:\n";
      for (let i = 0; i < todos.length; i++) {
        list += (i + 1) + ". " + todos[i] + "\n";
      }
      alert(list);
      console.log(list);
    }

  } else if (command === "delete") {
    if (todos.length === 0) {
      alert("Nothing to delete.");
    } else {
      let list = "Which task to delete?\n";
      for (let i = 0; i < todos.length; i++) {
        list += (i + 1) + ". " + todos[i] + "\n";
      }
      let index = prompt(list + "Enter number:");
      let num = Number(index);

      if (num >= 1 && num <= todos.length) {
        let removed = todos.splice(num - 1, 1);
        alert("Deleted: " + removed[0]);
        console.log("Deleted:", removed[0]);
      } else {
        alert("Invalid number.");
      }
    }

  } else if (command === "quit") {
    alert("Thanks for using Todo App!");
    running = false;

  } else {
    alert("Unknown command. Use: add, list, delete, quit");
  }
}

Concepts Used

  • Variables: todos array, running flag
  • Arrays: push, splice, length, indexing
  • Conditionals: if / else if / else chains
  • Loops: while for the app, for to display list
  • I/O: prompt, alert, console.log
  • Strings: trim, toLowerCase, concatenation
